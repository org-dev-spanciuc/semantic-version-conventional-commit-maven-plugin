3   /**
4    * Utility class for parameters sanitization.
5    */
6   public class ParameterSanitizer {
7   
8       /**
9        * To prevent instance creation.
10       */
11      private ParameterSanitizer() {
12      }
13  
14      /**
15       * Sanitizes a required string parameter.
16       * <p>
17       * Checks if value is not null and does not contain only white spaces. If so, removes leading
18       * and trailing white spaces and return the value.
19       *
20       * @param value     value to sanitize.
21       * @param valueName the parameter name.
22       * @return sanitized value.
23       * @throws IllegalArgumentException if the value is null.
24       * @throws IllegalArgumentException if the value contains only white spaces.
25       */
26      public static String sanitizeRequiredStringParameter(String value, String valueName) {
27          if (null == value) {
28              throw new IllegalArgumentException(
29                      String.format(Messages.MESSAGE_VALUE_MUST_NOT_BE_NULL, valueName));
30          }
31          String result = value.strip();
32          if (result.isEmpty()) {
33              throw new IllegalArgumentException(
34                      String.format(Messages.MESSAGE_VALUE_MUST_NOT_BE_EMPTY, valueName));
35          }
36          return result;
37      }
38  }
